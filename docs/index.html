Quick Example

local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/ijosephyusufk-dev/VoidXV2/main/VoidLib.lua"))()

local Window = Library:CreateWindow({
    Name = "My Script",
    Subtitle = "v1.0.0",
    LoadingEnabled = true
})

local MainTab = Window:CreateTab("Main", "rbxassetid://7734053495")

MainTab:CreateToggle("Auto Farm", false, function(state)
    print("Auto Farm:", state)
end)

MainTab:CreateSlider("Walk Speed", 16, 500, 16, function(value)
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = value
end)
