Components

VoidxV2 provides a rich set of UI components for building your script interface.

Toggle

local toggle = Tab:CreateToggle("Auto Farm", false, function(state)
    print("Toggle:", state)
end)

-- Methods
toggle:Set(true)     -- Set state
toggle:Get()         -- Get current state

Button

Tab:CreateButton("Click Me", function()
    print("Button clicked!")
end)

Slider

local slider = Tab:CreateSlider("Walk Speed", 16, 500, 16, function(value)
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = value
end)

-- Methods
slider:Set(100)      -- Set value
slider:Get()         -- Get current value

Dropdown

local dropdown = Tab:CreateDropdown("Select Zone", {"Zone 1", "Zone 2", "Zone 3"}, function(selected)
    print("Selected:", selected)
end)

-- Methods
dropdown:Set("Zone 2")           -- Set selection
dropdown:Get()                   -- Get current selection
dropdown:Refresh({"New", "List"})-- Update options

TextInput

local input = Tab:CreateInput("Username", "Enter name...", function(text)
    print("Input:", text)
end)

-- Methods
input:Set("new text")  -- Set text
input:Get()            -- Get current text

Keybind

Tab:CreateKeybind("Toggle ESP", Enum.KeyCode.E, function()
    print("Keybind pressed!")
end)

ColorPicker

local picker = Tab:CreateColorPicker("ESP Color", Color3.fromRGB(255, 0, 0), function(color)
    print("Color:", color)
end)

-- Methods
picker:Set(Color3.new(0, 1, 0))  -- Set color
picker:Get()                      -- Get current color
