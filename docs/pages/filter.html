Filter

Filter players based on distance, team, alive status, and custom conditions.

CreateFilter

local filter = Library.Developer:CreateFilter({
    Source = game.Players:GetPlayers(),
    Conditions = {
        MaxDistance = 100,    -- Within 100 studs
        TeamCheck = true,     -- Different team only
        ExcludeSelf = true,   -- Exclude local player
        AliveOnly = true      -- Only alive players
    }
})

local nearbyEnemies = filter:Apply()

Available Conditions

Condition Type Description
MaxDistance number Maximum distance in studs
TeamCheck boolean Only different team players
ExcludeSelf boolean Exclude local player
AliveOnly boolean Only players with Health > 0

Methods

local results = filter:Apply()       -- Apply filter and get results
local nearest = filter:GetNearest()  -- Get single nearest matching player

Practical Examples

-- Get nearby enemies
local enemies = Library.Developer:CreateFilter({
    Source = game.Players:GetPlayers(),
    Conditions = {
        MaxDistance = 50,
        TeamCheck = true,
        AliveOnly = true
    }
}):Apply()

print(#enemies .. " enemies nearby!")

-- Find nearest target
local target = Library.Developer:CreateFilter({
    Source = game.Players:GetPlayers(),
    Conditions = {
        ExcludeSelf = true,
        AliveOnly = true
    }
}):GetNearest()

if target then
    print("Nearest player:", target.Name)
end
